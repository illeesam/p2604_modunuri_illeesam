**핵심 포인트**:
- **모바일 앱 = 결제창만 InAppBrowser로 띄움**, 결과는 **딥링크(Custom URL Scheme)** 로 받음
- 웹과 백엔드 코드는 거의 동일 (PG사 SDK는 백엔드에서 호출)
- FO 앱만 결제 필요 — BO 앱은 딥링크 등록 불필요

`OdPayKakaoService.java`:

```java
@Service
@RequiredArgsConstructor
@Transactional(readOnly = true)
public class OdPayKakaoService {

    private final RestTemplate restTemplate;

    @Value("${kakao.pay.admin-key}")
    private String adminKey;

    @Value("${kakao.pay.cid}")
    private String cid;            // 일반: TC0ONETIME, 정기결제: TCSUBSCRIP

    private static final String KAKAO_HOST = "https://kapi.kakao.com";

    public Map<String, Object> ready(Map<String, Object> p) {
        Long orderId = Long.valueOf(p.get("orderId").toString());
        // 주문 조회 / 검증 ...

        HttpHeaders headers = new HttpHeaders();
        headers.set("Authorization", "KakaoAK " + adminKey);
        headers.setContentType(MediaType.APPLICATION_FORM_URLENCODED);

        MultiValueMap<String, String> body = new LinkedMultiValueMap<>();
        body.add("cid", cid);
        body.add("partner_order_id", String.valueOf(orderId));
        body.add("partner_user_id", p.get("memberId").toString());
        body.add("item_name", (String) p.get("itemName"));
        body.add("quantity", String.valueOf(p.get("quantity")));
        body.add("total_amount", String.valueOf(p.get("totalAmount")));
        body.add("tax_free_amount", "0");

        // ★ 모바일 앱: 딥링크 콜백 URL 사용
        body.add("approval_url", "shopjoy-fo://pay/success?orderId=" + orderId);
        body.add("fail_url",     "shopjoy-fo://pay/fail?orderId=" + orderId);
        body.add("cancel_url",   "shopjoy-fo://pay/cancel?orderId=" + orderId);

        ResponseEntity<Map> resp = restTemplate.postForEntity(
            KAKAO_HOST + "/v1/payment/ready",
            new HttpEntity<>(body, headers), Map.class);

        Map<String, Object> kakaoResp = resp.getBody();
        // 결과: { tid, next_redirect_mobile_url, next_redirect_app_url, ... }

        // tid 저장 (approve 시 필요)
        // odPayMapper.updateTid(orderId, (String) kakaoResp.get("tid"));

        Map<String, Object> result = new HashMap<>();
        result.put("tid", kakaoResp.get("tid"));
        result.put("redirectUrl", kakaoResp.get("next_redirect_mobile_url"));
        return result;
    }

    @Transactional
    public Map<String, Object> approve(Map<String, Object> p) {
        // pg_token, tid, partner_order_id, partner_user_id 으로 approve 호출
        // ...
        return Map.of();
    }
}
```

## Step 3. 결제창 호출 (FO 앱)

`utils/coPay.js` (신규 — FO 앱 전용):

```javascript
/**
 * coPay.js — Capacitor 모바일 결제 헬퍼
 *
 * 의존: utils/coNative.js, utils/foApiAxios.js
 *
 * 사용:
 *   await coPay.kakao({ orderId: 123, itemName: '상품A', quantity: 1, totalAmount: 10000 });
 *   await coPay.toss({ orderId: 123, amount: 10000, orderName: '상품A' });
 */
(function (global) {
  'use strict';

  async function openExternal(url) {
    if (global.coNative && global.coNative.isNative) {
      const Browser = global.coNative.plugin('Browser');
      if (Browser) {
        await Browser.open({ url, presentationStyle: 'popover' });
        return;
      }
    }
    // 웹 환경: 새 창 / 동일 창 이동
    global.open(url, '_blank');
  }

  async function closeExternal() {
    if (global.coNative && global.coNative.isNative) {
      const Browser = global.coNative.plugin('Browser');
      if (Browser && Browser.close) await Browser.close();
    }
  }

  /**
   * 카카오페이 결제 시작
   */
  async function kakao(payload) {
    const res = await global.foApi.post('/fo/od/pay/kakao/ready', payload,
      global.coUtil.apiHdr('주문', '카카오페이결제'));
    const { redirectUrl, tid } = res.data && res.data.data ? res.data.data : {};
    if (!redirectUrl) throw new Error('카카오페이 redirectUrl 응답 없음');
    // tid 는 세션에 저장 (approve 단계에서 사용)
    sessionStorage.setItem('kakaoPayTid', tid);
    sessionStorage.setItem('kakaoPayOrderId', String(payload.orderId));
    await openExternal(redirectUrl);
  }

  /**
   * 토스페이먼츠 결제 시작
   * (앱: 결제 위젯 URL을 백엔드에서 발급해서 사용 / 웹: SDK 직접 호출)
   */
  async function toss(payload) {
    const res = await global.foApi.post('/fo/od/pay/toss/ready', payload,
      global.coUtil.apiHdr('주문', '토스결제'));
    const data = res.data && res.data.data ? res.data.data : {};
    sessionStorage.setItem('tossOrderId', data.orderId);
    if (data.redirectUrl) {
      await openExternal(data.redirectUrl);
    } else {
      throw new Error('토스 redirectUrl 응답 없음 (백엔드 ready API 확인)');
    }
  }

  /**
   * 결제 완료 후 승인 처리
   * @param {String} pg - 'kakao' | 'toss'
   * @param {Object} params - 콜백 URL 쿼리 파라미터
   */
  async function approve(pg, params) {
    if (pg === 'kakao') {
      const tid = sessionStorage.getItem('kakaoPayTid');
      const orderId = sessionStorage.getItem('kakaoPayOrderId');
      const res = await global.foApi.post('/fo/od/pay/kakao/approve', {
        tid, orderId: Number(orderId), pgToken: params.pg_token,
      }, global.coUtil.apiHdr('주문', '카카오페이승인'));
      sessionStorage.removeItem('kakaoPayTid');
      sessionStorage.removeItem('kakaoPayOrderId');
      return res.data && res.data.data;
    }
    if (pg === 'toss') {
      const res = await global.foApi.post('/fo/od/pay/toss/approve', {
        paymentKey: params.paymentKey,
        orderId: params.orderId,
        amount: Number(params.amount),
      }, global.coUtil.apiHdr('주문', '토스승인'));
      sessionStorage.removeItem('tossOrderId');
      return res.data && res.data.data;
    }
    throw new Error(`알 수 없는 PG: ${pg}`);
  }

  global.coPay = { kakao, toss, approve, closeExternal };

  console.log('[coPay] 초기화 완료');
})(typeof window !== 'undefined' ? window : this);
```

## Step 4. 결제 완료 콜백 처리

앱이 딥링크(`shopjoy-fo://pay/success?...`)로 호출되면 Capacitor `App` 플러그인의 `appUrlOpen` 이벤트가 발생합니다.

`base/foApp.js` 마운트 코드 끝부분 또는 별도 `utils/coDeepLink.js`:

```javascript
/**
 * coDeepLink.js — 딥링크 처리 (FO 앱)
 *
 * 등록 위치: 앱 부팅 시 1회 (예: base/foApp.js 의 onMounted)
 */
(function (global) {
  'use strict';

  function parseUrl(url) {
    try {
      const u = new URL(url);
      const params = {};
      u.searchParams.forEach((v, k) => { params[k] = v; });
      return { host: u.hostname, path: u.pathname, params };
    } catch (e) {
      console.error('[coDeepLink] URL parse 실패', url, e);
      return null;
    }
  }

  async function handle(event) {
    const parsed = parseUrl(event.url);
    if (!parsed) return;
    const { host, path, params } = parsed;
    console.log('[coDeepLink] received', { host, path, params });

    // shopjoy-fo://pay/success | fail | cancel
    if (host === 'pay') {
      try { await (global.coPay && global.coPay.closeExternal && global.coPay.closeExternal()); } catch (e) {}

      const action = path.replace('/', '');
      if (action === 'success') {
        const orderId = params.orderId;
        const pg = params.pg_token ? 'kakao' : (params.paymentKey ? 'toss' : null);
        try {
          if (pg) await global.coPay.approve(pg, params);
          if (global.showToast) global.showToast('결제가 완료되었습니다.', 'success');
          if (global.navigate) global.navigate('myOrder', { dtlId: orderId });
        } catch (err) {
          console.error('[coDeepLink] approve 실패', err);
          if (global.showToast) global.showToast('결제 승인 실패: ' + (err.message || ''), 'error', 0);
        }
        return;
      }
      if (action === 'fail') {
        if (global.showToast) global.showToast('결제가 실패했습니다.', 'error');
        return;
      }
      if (action === 'cancel') {
        if (global.showToast) global.showToast('결제가 취소되었습니다.', 'error');
        return;
      }
    }

    // 그 외 일반 페이지 이동: shopjoy-fo://page/<pageId>?dtlId=...
    if (host === 'page' && global.navigate) {
      const pageId = path.replace('/', '');
      global.navigate(pageId, { dtlId: params.dtlId });
    }
  }

  function init() {
    if (!(global.coNative && global.coNative.isNative)) return;
    const App = global.coNative.plugin('App');
    if (!App) return;
    App.addListener('appUrlOpen', handle);
    console.log('[coDeepLink] 등록 완료');
  }

  global.coDeepLink = { init, handle };

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', init);
  } else {
    init();
  }
})(typeof window !== 'undefined' ? window : this);
```

## PG사별 차이점

| 항목 | 카카오페이 | 토스페이먼츠 | 토스페이(개인) |
|---|---|---|---|
| 모바일 결제창 호출 | redirect_url (백엔드에서 발급) | SDK 또는 redirect_url | redirect_url |
| 콜백 방식 | URL Scheme `shopjoy-fo://` | URL Scheme + Universal Link | URL Scheme |
| 결제 완료 검증 | `pg_token` + `tid` → approve API | `paymentKey` + `orderId` + `amount` → confirm API | `payToken` |
| 외부 앱 전환 | 카카오톡 / 카카오페이 앱 | 토스 앱 또는 카드사 앱 | 토스 앱 |
| 정기결제 | `cid=TCSUBSCRIP` + sid | 별도 빌링키 발급 API | 미지원 |
| 환불 | `/v1/payment/cancel` | `/v1/payments/{paymentKey}/cancel` | — |

## Universal Link / App Link (선택)

URL Scheme보다 보안 강화 — 앱 미설치 시 웹으로 폴백.

### Android App Link

`AndroidManifest.xml`:

```xml
<intent-filter android:autoVerify="true">
    <action android:name="android.intent.action.VIEW" />
    <category android:name="android.intent.category.DEFAULT" />
    <category android:name="android.intent.category.BROWSABLE" />
    <data android:scheme="https" android:host="illeesam.netlify.app" android:pathPrefix="/app/fo" />
</intent-filter>
```

서버에 `https://illeesam.netlify.app/.well-known/assetlinks.json` 배치:

```json
[{
  "relation": ["delegate_permission/common.handle_all_urls"],
  "target": {
    "namespace": "android_app",
    "package_name": "com.shopjoy.fo",
    "sha256_cert_fingerprints": ["..."]
  }
}]
```

### iOS Universal Link

Apple Developer → Identifiers → 앱 → **Associated Domains** 추가 (`applinks:illeesam.netlify.app`).

서버에 `https://illeesam.netlify.app/.well-known/apple-app-site-association`:

```json
{
  "applinks": {
    "apps": [],
    "details": [{
      "appID": "TEAMID.com.shopjoy.fo",
      "paths": ["/app/fo/*"]
    }]
  }
}
```

> **결제 콜백 URL은 URL Scheme이 더 안정적**. Universal Link는 일반 페이지 이동에만 사용 권장.

| 증상 | 원인 | 해결 |

| Android: 카카오톡으로 안 넘어감 | `LSApplicationQueriesSchemes` (iOS만 해당) | Android는 자동, iOS는 명시 필요 |
| iOS: `kakaopay` 앱이 안 열림 | `LSApplicationQueriesSchemes` 누락 | Info.plist 추가 |
| 결제 후 앱 복귀했지만 `appUrlOpen` 이벤트 미발생 | 리스너 등록 시점 누락 | `App.addListener('appUrlOpen', ...)` 가 부팅 직후 등록되었는지 확인 |
| `approve` 시 401 | tid/pgToken 만료 (10분) | 결제창 진입 후 10분 이내 처리 |
| 결제창에서 인증서 SSL 에러 | InAppBrowser의 Cleartext / Pinning 충돌 | `Browser.open` 의 `presentationStyle: 'popover'` 사용 |
| 안드로이드 백 버튼 시 결제창 닫히고 앱도 종료 | `backButton` 핸들러 없음 | `coNative.bindBackButton` 으로 인앱브라우저 닫기 처리 |
